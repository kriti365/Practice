pipeline
{
	agent any
 	stages
	{
		stage('Build artifacts')
		{
		steps
		{
		   sh 'mvn -f App/pom.xml clean package'
		}
			post
			{
			success
			{
			echo "Archiving the atifacts msg........."
			archiveArtifacts artifacts: '**/*.jar'
			}
			}
		}

		stage('Create Tomcat Docker Image')
		{
		steps
		{
		sh "docker build . -t dockerImage1:{env.BUILD_ID}
		}
		}
		
	}
}
